\frame{
\frametitle{Phase field modeling}
\begin{columns}[T] % align columns
\begin{column}{.47\textwidth}
Cahn-Hilliard dynamics
 \begin{equation*}
\Pi(c, \grad  c) = \int_{\Omega}  \left[  f( c ) + \grad  c  \cdot \kappa(\grad c) \grad  c  \right]  dV 
\end{equation*}
Chemical potential:
\begin{align*}
	   \v{\mu} = \delta_{c}\Pi( c , \grad  c)  
\end{align*}   
Kinetics:
\begin{align*}
   \frac{\partial c }{\partial t} = \grad \cdot (-\bL (\grad c) \grad \v{\mu}) 
\end{align*}  
\begin{itemize}
\item Models evolution of conserved fields like composition.
\item Fourth order PDE with complex anisotropic dependencies.
\end{itemize} 
%
\end{column}%
\hfill%
\begin{column}{.47\textwidth}
Allen-Cahn dynamics
\begin{equation*}
\Pi(\eta_i, \grad  \eta_i) = \int_{\Omega}  \left[  f( \eta_i ) +  \grad  \eta_i  \cdot \kappa (\grad  \eta_i) \grad  \eta_i  \right]  dV 
\end{equation*}
Chemical potential:
\begin{align*}
	   \v{\mu} = \delta_{\eta_i}\Pi( \eta_i, \grad  \eta_i)  
\end{align*} 
Kinetics:
\begin{align*}
   \frac{\partial \eta_i }{\partial t} = - (\bL (\grad \eta_i) ~ \v{\mu}    ) 
\end{align*}  
\begin{itemize}
\item Models evolution of non-conserved fields like structural order parameters. 
\item System of highly coupled second order PDE's.
\end{itemize} 
%
\end{column}%
\end{columns}

\small{\textit{Cahn ~\& Hilliard, J. Chem. Phys., 1958}}

\frame{
  \frametitle{Soft packing: A novel phase field approach}
 Let $\Omega\in \mathbb{R}^2$ with a smooth boundary $\partial \Omega$. Scalar fields $c_k, \;k = 1,\dots,N$ with $c_k \in [0,1]$ serve to delineate the interior and exterior of the cell numbered $k$. Here, the interior of cell $k$ is $\omega_k \subset \Omega$, where $\omega_k = \{ \bX \in \Omega\vert c_k(\bX) =1\}$. The exterior is $\Omega\backslash\omega_k$. The free energy density function is built up beginning with the following form: 
 \vspace{0.1in}
\begin{equation*}
\psi_1(c_k) = \alpha c_k^2(c_k-1)^2 + \frac{\kappa}{2}\vert\grad c_k\vert^2
\label{equ:psi1}
\end{equation*}\\
\vspace{0.2in}
The total free energy of the multi-cell aggregate is a functional $\Pi[\bc]$, defined as
\begin{align*}
\Pi[\bc] &:= \int\limits_\Omega \psi(\bc,\nabla c) ~\text{d}V\nonumber\\
&=\int\limits_\Omega \left(\sum_{k=1}^{N} f(c_k) + \sum_{k=1}^{N} \frac{\kappa}{2}\vert\grad c_k\vert^2 + \sum_{l\ne k}\sum_{k=1}^{N}\lambda c_k^2 c_l^2 \right) ~\text{d}V.
\label{equ:energy}
\end{align*}
}

\frame{
  \frametitle{Variational formulation}
  Taking the variational derivative with respect to $c_k$ yields
\begin{align*}
\delta \Pi_k[\bc;w] =  & \left.\frac{\text{d}}{\text{d}\epsilon} \int\limits_{\Omega} \sum_{k=1}^{N} \left(f(c_k+\epsilon w) + \frac{\kappa}{2}\vert\grad (c_k+\epsilon w)\vert^2+ \sum_{l\ne k}\lambda (c_k+\epsilon w)^2 c_l^2\right)  ~\text{d}V \right|_{\epsilon=0} \nonumber\\
=&\int\limits_{\Omega} w \left( f^\prime(c_k) -  \kappa \Delta  c_k  + \sum_{l\ne k}2\lambda  c_k c_l^2 \right) ~dV + \int\limits_{\partial \Omega}   w \kappa \grad c_k \cdot \bn   ~dS
\end{align*} \\
\vspace{0.1in}
The chemical potential of the $k^\text{th}$ cell is identified as,
\begin{equation*}
\mu_k  = f^\prime(c_k) -  \kappa \Delta c_k + \sum_{l\ne k}2\lambda  c_k c_l^2
\label{equ:chemo_potential}
\end{equation*}\\
\vspace{0.1in}
Resulting kinetics:
\begin{equation*}
\frac{\partial c_k}{\partial t} = -~\grad \cdot (-M\grad \mu_k) + s_k
\label{equ:dynamic}
\end{equation*}
}

\frame{
  \frametitle{Variational formulation}
 Time discretization:
\begin{align*}
c^{n+1}_{k} &= c^{n}_{k} + \Delta t (M~\grad \cdot (\grad \mu^{n+1}_{k})+s_{k})\nonumber\\
\text{where}\quad\mu^{n+1}_{k} &= {f^\prime}^ {n+1}(c_k) -  \kappa \Delta c^{n+1}_{k} + \sum_{l\ne k}2\lambda  c_k^{n+1} {c_l^{n+1}}^2
\end{align*}\\
\vspace{0.1in}
Weak formulation:
\begin{align*}
\int\limits_{\Omega}   w c_k^{n+1} ~\text{d}V
&= \int\limits_{\Omega}   (w  c_k^{n} - \grad w \cdot \Delta t M \grad \mu_k^{n+1} + w\Delta t s_k) ~\text{d}V\nonumber\\
\int\limits_{\Omega}   w  \mu_{k}^{n+1}  ~\text{d}V 
&=\int\limits_{\Omega}  (w  {f^\prime}^{n+1}(c_k) + \grad w \cdot \kappa \grad c_k^{n+1} ~dV + \int\limits_\Omega w \sum_{l\ne k}2\lambda  c_k^{n+1} {c_l^{n+1}}^2)  ~\text{d}V
\end{align*}\\
\vspace{0.2in}
Implemented in the \textit{deal.II} finite element framework.
}

\frame{
  \frametitle{Observations}
  Highlights of the phase-only representation of soft packing:
  \begin{itemize}
  \item Natural representation for cell-cell interactions like cell division, cell-cell contact and cell-cell adhesion*.
  \item No need for explicit topology tracking and topology changes, and model identical in 2D/3D.
  \item Diffusion provides a good presentation for cell mobility. Optimal packing solutions depend on the initial conditions and cell mobilities. 
  \end{itemize}
  \vspace{0.2in}
  Limitations: 
   \begin{itemize}
  \item Lacks explicit representation for mechanics. No underlying material model to represent the cytoskeletal mechanics or cell-cell interactions. 
  \item Problem size scales with the number of cell considered. 
  \item Phase field related length and time scales limit space and time discretization. 
 \end{itemize}
 \vspace{0.2in}
  https://arxiv.org/abs/1806.01410
}


\frame{
  \frametitle{Phase and Shape model: Basic treatment of mechanics}
  Shape changes of a cell determined through the changes to the principle moment of inertia ($I^1$, $I^2$) of the cell:
\begin{equation*}
\Pi_{\text{MI}} = \sum_{i=1}^{\text{dim}}\delta^i ({I_i} ^\text{ref}-I_i)^2
\end{equation*}
where $I_1$ and $I_2$ are obtained from the Moment of Inertia tensor: 
\begin{equation*}
\bI[c] = 
\begin{bmatrix}
I_{11}[c] & I_{12}[c] \\
I_{21}[c] & I_{22}[c]
\end{bmatrix}
=
\begin{bmatrix}
\int c \bar{X}_1^2 ~\text{d}V   & \int c\bar{X}_1 \bar{X}_2 ~\text{d}V\\
\int c\bar{X}_1 \bar{X}_2 ~\text{d}V & \int c\bar{X}_2^2 ~\text{d}V
\end{bmatrix}
\label{equ:MomentOfInertia}
\end{equation*}\\
\vspace{0.2in}
Modified free energy functional: 
\begin{align}
\Pi[\bc,\nabla\bc] &:= \int\limits_{\Omega} \left(\sum_{k=1}^{N} f(c_k) + \sum_{k=1}^{N} \frac{\kappa}{2}\vert\grad c_k\vert^2 + \sum_{l\ne k}\sum_{k=1}^{N}\lambda c_k^2 c_l^2 \right) ~\text{d}V\nonumber\\
&\phantom{+}+ \sum_{k=1}^{N}\sum_{i=1}^{\text{dim}}\delta_k^i ({I^k_i} ^\text{ref}-I^k_i)^2
\end{align}
}

\frame{
  \frametitle{Summary and look ahead}
  \begin{itemize}
  \setlength\itemsep{1em}
   \item Developed a diffused interface based numerical framework for modeling growth and packing of cell aggregates. Salient features: \vspace{0.05in}
  \begin{itemize}
  \setlength\itemsep{1em}
  \item No discreet interface evolving mechanisms like those employed in lattice, cell-centric and vertex dynamics models needed.
  \item Time evolution occurs at realistic time scales controlled by the growth rate, or doubling time of the cells, and does not need equilibrations needed by other discrete models.
  \item Any arbitrary cell shape can be represented without being limited to polygonal shapes or a jagged representation of the cell boundary.
  \item Basic model for incorporating mechanics and material models of the underlying cytoskeletal network. 
  \end{itemize}
  \item Ongoing numerical work on incorporating (1) A material model in place of the shape model, (2) Active parameter tracking to allow for scaling independent of the number of cells.
  \item Key applications: (1) Soft packing in cells, (2) Patterning in embryogenesis, (3) Other problems involving cell aggregates like collective cell motion. 
  \end{itemize}
